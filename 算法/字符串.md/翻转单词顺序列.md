 public String reverseWord(String word) {
        if (word == null && word.length() == 0) {
            return word;
        }

        char[] arr = word.toCharArray();
        int i = 0, j = 0;
        while (j <= arr.length) {
            char c = arr[j];
            if (j == arr.length || c == ' ') {
                reverse(arr, i, j - 1);
                i = j + 1;
            }
            j++;
        }
        reverse(arr, 0, arr.length - 1);
        return new String(arr);
    }

    private void reverse(char[] arr, int i, int j) {
        while (i < j) {
            char temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;

        }
    }
